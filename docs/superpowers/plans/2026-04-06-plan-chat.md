# Plan Chat Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a native plan creation experience that opens a Claude Code terminal for conversational brainstorming alongside a live plan preview webview with annotations.

**Architecture:** A `PlanChatSession` coordinates a VS Code integrated terminal (running `claude` CLI with a system prompt for plan-format output) and a `PlanPreviewPanel` webview (live-rendering PLAN.md via FileSystemWatcher + plan parser). Annotations in the preview inject text into the terminal via `terminal.sendText()`.

**Tech Stack:** TypeScript, VS Code Terminal API, VS Code Webview API, Vitest

**Spec:** `docs/superpowers/specs/2026-04-06-plan-chat-design.md`

---

### Design Decisions

- **Reuse existing `Detection` class** for Claude CLI detection — same constructor pattern, different binary path and minimum version.
- **New single-file watcher** for PLAN.md — follow `WatcherManager` debounce pattern but don't modify the existing class. A simple `vscode.workspace.createFileSystemWatcher` with debounce is sufficient.
- **Plan parser extension** — `parsePlan()` extracts phases but not descriptions. We extend it to also extract body text (lines between headers). New `validatePlan()` function checks sequentiality and dependency references.
- **System prompt as inline `--append-system-prompt`** — avoid file management by using the CLI flag directly. Verified: `claude --help` shows `--append-system-prompt <prompt>`.
- **Terminal spawned as `shellPath`** — `vscode.window.createTerminal({ shellPath: claudePath, shellArgs: [...] })` gives us full control over the process.

```typescript
export async function detectClaude(
  executor: Executor,
  claudePath: string,
): Promise<IDetectionResult> {
  try {
    const { stdout } = await executor(claudePath, ["--version"]);
    const version = stdout.trim();
    return {
      status: "detected",
      path: claudePath,
      version,
      minimumVersion: "0.0.0",
    };
  } catch {
    return {
      status: "not-found",
      minimumVersion: "0.0.0",
    };
  }
}
```

```typescript
// src/parsers/planValidator.ts
import { type PlanState } from "../types";

export interface ValidationWarning {
  type: "non-sequential" | "dangling-dependency";
  message: string;
  phaseNumber: number | string;
}

export interface ValidationResult {
  valid: boolean;
  warnings: ValidationWarning[];
}

export function validatePlan(plan: PlanState): ValidationResult {
  const warnings: ValidationWarning[] = [];
  const phaseNumbers = new Set(plan.phases.map((p) => p.number));

  for (let i = 1; i < plan.phases.length; i++) {
    const prev = Number(plan.phases[i - 1].number);
    const curr = Number(plan.phases[i].number);
    if (!isNaN(prev) && !isNaN(curr) && curr !== prev + 1) {
      warnings.push({
        type: "non-sequential",
        message: `Phase ${curr} follows Phase ${prev} (expected ${prev + 1})`,
        phaseNumber: plan.phases[i].number,
      });
    }
  }

  for (const phase of plan.phases) {
    if (!phase.dependencies) continue;
    for (const dep of phase.dependencies) {
      const depNum = Number(dep);
      if (!isNaN(depNum) && !phaseNumbers.has(depNum)) {
        warnings.push({
          type: "dangling-dependency",
          message: `Phase ${phase.number} depends on Phase ${dep}, which does not exist`,
          phaseNumber: phase.number,
        });
      }
    }
  }

  return { valid: warnings.length === 0, warnings };
}
```

```typescript
// src/parsers/planDescription.ts
import { parsePlan } from "./plan";
import { type PlanState, type PlanPhase } from "../types";

export interface PlanPhaseWithDescription extends PlanPhase {
  description: string;
}

export interface PlanStateWithDescriptions {
  phases: PlanPhaseWithDescription[];
}

const STATUS_RE = /^\[status:\s*[^\]]+\]/i;
const DEPENDS_RE = /^\*\*Depends on:\*\*/;

export function parsePlanWithDescriptions(content: string): PlanStateWithDescriptions {
  const plan = parsePlan(content);
  const lines = content.split("\n");

  return {
    phases: plan.phases.map((phase, i) => {
      const startLine = phase.headerLine + 1;
      const endLine = phase.bodyEndLine;
      const bodyLines = lines
        .slice(startLine, endLine + 1)
        .filter((line) => !STATUS_RE.test(line.trim()) && !DEPENDS_RE.test(line.trim()));

      const description = bodyLines.join("\n").trim();
      return { ...phase, description };
    }),
  };
}
```

```typescript
// src/views/planPreviewPanel.ts
import { randomBytes } from "crypto";
import { renderPlanPreviewShell, renderPhaseCardsHtml } from "./planPreviewHtml";
import { parsePlanWithDescriptions } from "../parsers/planDescription";
import { validatePlan } from "../parsers/planValidator";
import { parsePlan } from "../parsers/plan";

// Webview and WebviewPanel interfaces — same as liveRunPanel.ts

export interface PlanPreviewPanelDeps {
  createWebviewPanel: (viewType: string, title: string, showOptions: number, options: { enableScripts: boolean; retainContextWhenHidden: boolean }) => WebviewPanel;
  readFile: (path: string) => Promise<string>;
  onAnnotation: (phase: number | string, title: string, text: string) => void;
}

export class PlanPreviewPanel {
  private _panel: WebviewPanel | undefined;
  private _deps: PlanPreviewPanelDeps;
  private _sessionActive = true;

  constructor(deps: PlanPreviewPanelDeps) { this._deps = deps; }

  reveal(): void {
    if (this._panel) { this._panel.reveal(); return; }
    const nonce = randomBytes(16).toString("hex");
    this._panel = this._deps.createWebviewPanel(
      "oxveil.planPreview", "Plan Preview", 2,
      { enableScripts: true, retainContextWhenHidden: true },
    );
    this._panel.webview.html = renderPlanPreviewShell(nonce, this._panel.webview.cspSource);
    this._panel.onDidDispose(() => { this._panel = undefined; });
    this._panel.webview.onDidReceiveMessage((msg: any) => {
      if (msg.type === "annotation") {
        this._deps.onAnnotation(msg.phase, msg.title, msg.text);
      }
    });
  }

  async onFileChanged(filePath: string): Promise<void> {
    if (!this._panel) return;
    let content: string;
    try { content = await this._deps.readFile(filePath); }
    catch { this._panel.webview.postMessage({ type: "phases", phases: [], validation: { valid: true, warnings: [] } }); return; }

    const plan = parsePlan(content);
    const planWithDesc = parsePlanWithDescriptions(content);
    const validation = validatePlan(plan);

    if (plan.phases.length === 0) {
      this._panel.webview.postMessage({ type: "phases", phases: [], validation, rawContent: content, sessionActive: this._sessionActive });
    } else {
      this._panel.webview.postMessage({ type: "phases", phases: planWithDesc.phases, validation, sessionActive: this._sessionActive });
    }
  }

  setSessionActive(active: boolean): void {
    this._sessionActive = active;
    this._panel?.webview.postMessage({ type: "session-state", active });
  }

  dispose(): void { this._panel?.dispose(); this._panel = undefined; }
}
```

```typescript
// src/core/planChatSession.ts
import * as path from "path";

interface Terminal {
  show: () => void;
  sendText: (text: string) => void;
  dispose: () => void;
}

interface FileWatcher {
  onDidChange: (cb: (uri: { fsPath: string }) => void) => void;
  onDidCreate: (cb: (uri: { fsPath: string }) => void) => void;
  onDidDelete: (cb: (uri: { fsPath: string }) => void) => void;
  dispose: () => void;
}

interface PanelLike {
  reveal: () => void;
  onFileChanged: (path: string) => Promise<void>;
  setSessionActive: (active: boolean) => void;
  dispose: () => void;
}

export interface PlanChatSessionDeps {
  createTerminal: (options: { name: string; cwd: string; shellPath: string; shellArgs: string[] }) => Terminal;
  createFileWatcher: (glob: string) => FileWatcher;
  planPreviewPanel: PanelLike;
  onSessionEnd: () => void;
}

export interface StartOptions {
  claudePath: string;
  workspaceRoot: string;
  folderName: string;
  systemPrompt: string;
}

export class PlanChatSession {
  private _deps: PlanChatSessionDeps;
  private _terminal: Terminal | undefined;
  private _watcher: FileWatcher | undefined;
  private _active = false;
  private _debounceTimer: ReturnType<typeof setTimeout> | undefined;

  constructor(deps: PlanChatSessionDeps) { this._deps = deps; }

  start(options: StartOptions): void {
    const { claudePath, workspaceRoot, folderName, systemPrompt } = options;
    const planPath = path.join(workspaceRoot, "PLAN.md");

    this._terminal = this._deps.createTerminal({
      name: `Plan Chat (${folderName})`,
      cwd: workspaceRoot,
      shellPath: claudePath,
      shellArgs: ["--append-system-prompt", systemPrompt, "--permission-mode", "plan"],
    });
    this._terminal.show();

    const glob = path.join(workspaceRoot, "PLAN.md");
    this._watcher = this._deps.createFileWatcher(glob);
    const handleChange = (uri: { fsPath: string }) => {
      if (this._debounceTimer) clearTimeout(this._debounceTimer);
      this._debounceTimer = setTimeout(() => {
        this._deps.planPreviewPanel.onFileChanged(uri.fsPath);
      }, 200);
    };
    this._watcher.onDidChange(handleChange);
    this._watcher.onDidCreate(handleChange);

    this._deps.planPreviewPanel.reveal();
    this._active = true;

    // Initial load if PLAN.md already exists
    this._deps.planPreviewPanel.onFileChanged(planPath);
  }

  sendAnnotation(phase: number | string, title: string, text: string): void {
    if (!this._terminal) return;
    this._terminal.sendText(`Regarding Phase ${phase} (${title}): ${text}`);
  }

  isActive(): boolean { return this._active; }

  dispose(): void {
    if (this._debounceTimer) clearTimeout(this._debounceTimer);
    this._terminal?.dispose();
    this._watcher?.dispose();
    this._terminal = undefined;
    this._watcher = undefined;
    this._active = false;
    this._deps.planPreviewPanel.setSessionActive(false);
    this._deps.onSessionEnd();
  }
}
```

```typescript
// src/commands/planChat.ts
import * as path from "path";

const SYSTEM_PROMPT_TEMPLATE = `You are helping the user create a plan for an AI coding workflow.

Write the plan to PLAN.md in the workspace root. Use this exact format:

## Phase N: Title
Description of what this phase accomplishes.
**Depends on:** Phase X, Phase Y

Rules:
- Number phases sequentially starting from 1
- Include clear descriptions under each phase header
- Declare dependencies explicitly
- Update PLAN.md after each refinement (overwrite, don't append)
- Keep phase count reasonable (3-15 phases for most projects)`;

export function buildSystemPrompt(existingContent?: string): string {
  if (!existingContent) return SYSTEM_PROMPT_TEMPLATE;
  return `${SYSTEM_PROMPT_TEMPLATE}\n\nThe user has an existing plan. Here is the current content:\n\n${existingContent}\n\nHelp them refine this plan.`;
}

export async function handleExistingPlan(
  planPath: string,
  readFile: (path: string) => Promise<string>,
): Promise<{ action: "none" | "exists"; content?: string }> {
  try {
    const content = await readFile(planPath);
    return { action: "exists", content };
  } catch {
    return { action: "none" };
  }
}
```
